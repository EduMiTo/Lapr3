CREATE TABLE Ship (mmsi varchar2(9) NOT NULL, imo varchar2(10) NOT NULL UNIQUE, callSign varchar2(255) NOT NULL UNIQUE, numEnergyGenerators number(10), GeneratorType varchar2(255), draft float(10) NOT NULL, name varchar2(255) NOT NULL UNIQUE, vesselType number(10) NOT NULL, width float(10) NOT NULL, lenght float(10) NOT NULL, capacity float(10), shipCaptain varchar2(255), PRIMARY KEY (mmsi));
CREATE TABLE Container (Id number(10), isoCode varchar2(255) NOT NULL, checkDigit number(10) NOT NULL, maxweight float(10) NOT NULL, weight float(10) NOT NULL, maxweightpacked float(10) NOT NULL, maxvolumepacked float(10) NOT NULL, repairrecommendation varchar2(255) NOT NULL, certificates varchar2(255) NOT NULL, payload float(10) NOT NULL, tare float(10) NOT NULL, gross float(10) NOT NULL, PRIMARY KEY (Id));
CREATE TABLE CargoManifestLoad (id number(10), Portid number(10), Shipmmsi varchar2(9), isConcluded number(10), Warehouseid number(10), Truckregistration varchar2(255), PRIMARY KEY (id));
CREATE TABLE Country (name varchar2(255) NOT NULL, continent varchar2(255),population float(10), alpha2_code varchar2(255), alpha3_code varchar(255), PRIMARY KEY (name));
CREATE TABLE ShipPosition (shipmmsi varchar2(9) NOT NULL, dateaismessage timestamp(0) NOT NULL, latitude varchar2(255) NOT NULL, longitude varchar2(255) NOT NULL, sog float(10) NOT NULL, cog float(10) NOT NULL, heading varchar2(255) NOT NULL, position number(10), cargo varchar2(255) NOT NULL, PRIMARY KEY (shipmmsi, dateaismessage));
CREATE TABLE Truck (registration varchar2(255) NOT NULL, PRIMARY KEY (registration));
CREATE TABLE CargoManifest_Container (cargomanifestloadId number(10) NOT NULL, containerId number(10) NOT NULL, TripId number(10) NOT NULL, xcontainer number(10) NOT NULL, ycontainer number(10) NOT NULL, zcontainer number(10) NOT NULL, grossContainer float(10) NOT NULL, TripCargoManifestLoadid number(10) NOT NULL, CargoManifestUnloadId number(10), changesResponsible varchar2(255), clientOwner varchar2(255), PRIMARY KEY (cargomanifestloadId, containerId));
CREATE TABLE Warehouse (id number(10), name varchar2(255) NOT NULL, placeLocationlatitude varchar2(255) NOT NULL, placeLocationlongitude varchar2(255) NOT NULL, capacity number(10), ocuppancyRate float(10), portId Number(10),  PRIMARY KEY (id));
CREATE TABLE PlaceLocation (latitude varchar2(255) NOT NULL, longitude varchar2(255) NOT NULL, countryname varchar2(255) NOT NULL, PRIMARY KEY (latitude, longitude));
CREATE TABLE Trip (CargoManifestLoadid number(10) NOT NULL, id number(10) NOT NULL, Origin varchar2(255), Destination varchar2(255), expectedDepartureDate timestamp(0), expectedArrivalDate timestamp(0), realDepartureDate timestamp(0), realArrivalDate timestamp(0), PRIMARY KEY (CargoManifestLoadid, id));
CREATE TABLE CargoManifestUnload (Id number(10), TripId number(10) NOT NULL, CargoManifestLoadid number(10) NOT NULL, Portid number(10), Shipmmsi varchar2(9), WarehouseId number(10), PRIMARY KEY (Id));
CREATE TABLE Port (id number(10), name varchar2(255), PlaceLocationlatitude varchar2(255) NOT NULL, PlaceLocationlongitude varchar2(255) NOT NULL, capacity number(10) NOT NULL, ocuppancyRate float(10),capacityShip Number(10), PRIMARY KEY (id));
CREATE TABLE Port_Ship (Portid number(10) NOT NULL, Shipmmsi varchar2(9) NOT NULL, PRIMARY KEY (Portid, Shipmmsi));
CREATE TABLE Truck_Warehouse (Warehouseid number(10) NOT NULL, Truckregistration varchar2(255) NOT NULL, PRIMARY KEY (Warehouseid, Truckregistration));
CREATE TABLE Role (id number(10), designation varchar2(255) NOT NULL UNIQUE, PRIMARY KEY (id));
CREATE TABLE Users (username varchar2(255) NOT NULL, password varchar2(255) NOT NULL, Roleid number(10) NOT NULL, PRIMARY KEY (username));
CREATE TABLE AuditTrails (id number(10), Userusername varchar2(255), Operationsid number(10) NOT NULL, CargoManifest_ContainercargomanifestloadId number(10) NOT NULL, CargoManifest_ContainercontainerId number(10) NOT NULL,dateOfOP TIMESTAMP, PRIMARY KEY (id));
CREATE TABLE Operations (id number(10), operationName varchar2(6) NOT NULL UNIQUE, PRIMARY KEY (id));
CREATE TABLE Capital (name varchar2(255), Countryname varchar2(255) NOT NULL, PlaceLocationlatitude varchar2(255) NOT NULL, PlaceLocationlongitude varchar2(255) NOT NULL, PRIMARY KEY (name));
CREATE TABLE Border (Countryname varchar2(255) NOT NULL, Countryname2 varchar2(255) NOT NULL, PRIMARY KEY (Countryname, Countryname2));
CREATE TABLE SeaDist (PortidFrom number(10) NOT NULL, PortidTo number(10) NOT NULL, SeaDistance float(10), PRIMARY KEY (PortidFrom, PortidTo));
